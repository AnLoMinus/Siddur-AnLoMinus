מודה אני

מוֹדֶה אֲנִי לְפָנֶיךָ מֶלֶךְ חַי וְקַיָּם, שֶׁהֶחֱזַרְתָּ בִּי נִשְׁמָתִי בְּחֶמְלָה, רַבָּה אֱמוּנָתֶךָ.

הסבר:

תפילת הודיה קצרה הנאמרת מיד עם הקימה בבוקר, לפני נטילת ידיים. בתפילה זו אנו מודים לה' על שהחזיר לנו את נשמתנו בבוקר.

נטילת ידיים

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ עַל נְטִילַת יָדַיִם.

הסבר:

נטילת ידיים היא מצווה מהתורה, והיא נעשית עם הקימה בבוקר, לפני כל אכילת לחם, ולאחר השירותים.

אשר יצר

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר יָצַר אֶת הָאָדָם בְּחָכְמָה, וּבָרָא בוֹ נְקָבִים נְקָבִים, חֲלוּלִים חֲלוּלִים. גָּלוּי וְיָדוּעַ לִפְנֵי כִסֵּא כְבוֹדֶךָ, שֶׁאִם יִפָּתֵחַ אֶחָד מֵהֶם, אוֹ יִסָּתֵם אֶחָד מֵהֶם, אִי אֶפְשַׁר לְהִתְקַיֵּם וְלַעֲמוֹד לְפָנֶיךָ. בָּרוּךְ אַתָּה ה', רוֹפֵא כָל בָּשָׂר וּמַפְלִיא לַעֲשׂוֹת.

הסבר:

ברכה זו נאמרת לאחר היציאה מהשירותים, והיא מביעה את ההכרה בחכמת הבורא שברא את גוף האדם בצורה מופלאה.

אלוהי נשמה

אֱלֹהַי, נְשָׁמָה שֶׁנָּתַתָּ בִּי טְהוֹרָה הִיא. אַתָּה בְרָאתָהּ, אַתָּה יְצַרְתָּהּ, אַתָּה נְפַחְתָּהּ בִּי, אַתָּה מְשַׁמְּרָהּ בְּקִרְבִּי, וְאַתָּה עָתִיד לִטְּלָהּ מִמֶּנִּי, וּלְהַחֲזִירָהּ בִּי לֶעָתִיד לָבוֹא. כָּל זְמַן שֶׁהַנְּשָׁמָה בְּקִרְבִּי, מוֹדֶה אֲנִי לְפָנֶיךָ ה' אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי, רִבּוֹן כָּל הַמַּעֲשִׂים, אֲדוֹן כָּל הַנְּשָׁמוֹת. בָּרוּךְ אַתָּה ה', הַמַּחֲזִיר נְשָׁמוֹת לִגְוִיּוֹת מֵתִים.

הסבר:

תפילה זו נאמרת בבוקר, והיא מביעה את ההכרה בכך שהנשמה היא מתנה מה' והוא זה שמחזיר אותה לגוף בכל בוקר.

ברכות התורה

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ לַעֲסֹק בְּדִבְרֵי תוֹרָה.

וְהַעֲרֶב נָא ה' אֱלֹהֵינוּ אֶת דִּבְרֵי תוֹרָתְךָ בְּפִינוּ וּבְפִי עַמְּךָ בֵּית יִשְׂרָאֵל, וְנִהְיֶה אֲנַחְנוּ וְצֶאֱצָאֵינוּ וְצֶאֱצָאֵי עַמְּךָ בֵּית יִשְׂרָאֵל, כֻּלָּנוּ יוֹדְעֵי שְׁמֶךָ וְלוֹמְדֵי תוֹרָתְךָ לִשְׁמָהּ. בָּרוּךְ אַתָּה ה', הַמְלַמֵּד תּוֹרָה לְעַמּוֹ יִשְׂרָאֵל.

הסבר:

ברכות התורה נאמרות לפני לימוד תורה, ומביעות את ההכרה בחשיבות לימוד התורה ואת התפילה שהלימוד יהיה לשמה.

פסוקים לאחר ברכת התורה

וְאָהַבְתָּ אֵת ה' אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ וּבְכָל מְאֹדֶךָ.

וְהָיוּ הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם עַל לְבָבֶךָ.

וְשִׁנַּנְתָּם לְבָנֶיךָ וְדִבַּרְתָּ בָּם בְּשִׁבְתְּךָ בְּבֵיתֶךָ וּבְלֶכְתְּךָ בַדֶּרֶךְ וּבְשָׁכְבְּךָ וּבְקוּמֶךָ.

וּקְשַׁרְתָּם לְאוֹת עַל יָדֶךָ וְהָיוּ לְטֹטָפֹת בֵּין עֵינֶיךָ.

וּכְתַבְתָּם עַל מְזֻזוֹת בֵּיתֶךָ וּבִשְׁעָרֶיךָ.

הסבר:

פסוקים אלו, הלקוחים מספר דברים, מדגישים את החשיבות של אהבת ה' ושמירת המצוות בכל עת ובכל מקום.

אלו דברים שאין להם שיעור

אֵלּוּ דְבָרִים שֶׁאֵין לָהֶם שִׁעוּר: הַפֵּאָה, וְהַבִּכּוּרִים, וְהָרְאָיוֹן, וּגְמִילוּת חֲסָדִים, וְתַלְמוּד תּוֹרָה.

אֵלּוּ דְבָרִים שֶׁאָדָם אוֹכֵל פֵּרוֹתֵיהֶם בָּעוֹלָם הַזֶּה, וְהַקֶּרֶן קַיֶּמֶת לוֹ לָעוֹלָם הַבָּא: כִּבּוּד אָב וָאֵם, וּגְמִילוּת חֲסָדִים, וְהַשְׁכָּמַת בֵּית הַמִּדְרָשׁ שַׁחֲרִית וְעַרְבִית, וְהַכְנָסַת אוֹרְחִים, וּבִקּוּר חוֹלִים, וְהַכְנָסַת כַּלָּה, וּלְוָיַת הַמֵּת, וְעִיּוּן תְּפִלָּה, וַהֲבָאַת שָׁלוֹם בֵּין אָדָם לַחֲבֵרוֹ, וּבֵין אִישׁ לְאִשְׁתּוֹ. וְתַלְמוּד תּוֹרָה כְּנֶגֶד כֻּלָּם.

הסבר:

קטע זה ממסכת פאה מדגיש את חשיבותם של מעשים טובים ושל לימוד תורה, שאין להם שיעור, ואת השכר הגדול הצפוי להם בעולם הבא.

סדר ההליכה לבית הכנסת

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ לִהְיוֹת מִתְפַּלֵּל בְּצִבּוּר.

הִנְנִי מוּכָן וּמְזֻמָּן לְקַיֵּם מִצְוַת עֲשֵׂה שֶׁל תְּפִלָּה בְּצִבּוּר. יְהִי רָצוֹן מִלְּפָנֶיךָ ה' אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי, שֶׁתְּהֵא תְּפִלָּתִי בְּצִבּוּר מְקֻבֶּלֶת וּרְצוּיָה לְפָנֶיךָ.

הסבר:

תפילות אלו נאמרות בדרך לבית הכנסת, ומביעות את ההכנה הנפשית לתפילה בציבור ואת התפילה שהתפילה תהיה רצויה לפני ה'.

תיקון רחל

כֹּה אָמַר ה': קוֹל בְּרָמָה נִשְׁמָע נְהִי בְּכִי תַמְרוּרִים רָחֵל מְבַכָּה עַל בָּנֶיהָ מֵאֲנָה לְהִנָּחֵם עַל בָּנֶיהָ כִּי אֵינֶנּוּ.

כֹּה אָמַר ה': מִנְעִי קוֹלֵךְ מִבֶּכִי וְעֵינַיִךְ מִדִּמְעָה כִּי יֵשׁ שָׂכָר לִפְעֻלָּתֵךְ נְאֻם ה' וְשָׁבוּ מֵאֶרֶץ אוֹיֵב.

וְיֵשׁ תִּקְוָה לְאַחֲרִיתֵךְ נְאֻם ה' וְשָׁבוּ בָנִים לִגְבוּלָם.

הסבר:

תיקון רחל הוא חלק מסדר תיקון חצות, הנאמר בחצות הלילה. הוא מביע את צערה של רחל אמנו על גלות בניה, ואת הנחמה שה' מבטיח לה על שובם לארץ.

תיקון לאה

וַיַּרְא ה' כִּי שְׂנוּאָה לֵאָה וַיִּפְתַּח אֶת רַחְמָהּ וְרָחֵל עֲקָרָה.

וַתַּהַר לֵאָה וַתֵּלֶד בֵּן וַתִּקְרָא שְׁמוֹ רְאוּבֵן כִּי אָמְרָה כִּי רָאָה ה' בְּעָנְיִי כִּי עַתָּה יֶאֱהָבַנִי אִישִׁי.

וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר כִּי שָׁמַע ה' כִּי שְׂנוּאָה אָנֹכִי וַיִּתֶּן לִי גַּם אֶת זֶה וַתִּקְרָא שְׁמוֹ שִׁמְעוֹן.

הסבר:

תיקון לאה הוא החלק השני של סדר תיקון חצות. הוא מתאר את סיפור לידת בניה של לאה אמנו, ואת ההכרה בכך שה' רואה את מצוקתה ונותן לה בנים.

ברכות השחר

אֱלֹהַי, נְשָׁמָה שֶׁנָּתַתָּ בִּי טְהוֹרָה הִיא. אַתָּה בְרָאתָהּ, אַתָּה יְצַרְתָּהּ, אַתָּה נְפַחְתָּהּ בִּי, אַתָּה מְשַׁמְּרָהּ בְּקִרְבִּי, וְאַתָּה עָתִיד לִטְּלָהּ מִמֶּנִּי, וּלְהַחֲזִירָהּ בִּי לֶעָתִיד לָבוֹא. כָּל זְמַן שֶׁהַנְּשָׁמָה בְּקִרְבִּי, מוֹדֶה אֲנִי לְפָנֶיךָ ה' אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי, רִבּוֹן כָּל הַמַּעֲשִׂים, אֲדוֹן כָּל הַנְּשָׁמוֹת. בָּרוּךְ אַתָּה ה', הַמַּחֲזִיר נְשָׁמוֹת לִגְוִיּוֹת מֵתִים.

ברכות התורה

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ לַעֲסֹק בְּדִבְרֵי תוֹרָה.

פתיחת אליהו

וְהוּא רַחוּם יְכַפֵּר עָוֹן וְלֹא יַשְׁחִית וְהִרְבָּה לְהָשִׁיב אַפּוֹ וְלֹא יָעִיר כָּל חֲמָתוֹ.

סדר עטיפת ציצית

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ עַל מִצְוַת צִיצִית.

סדר הנחת תפילין

בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, אֲשֶׁר קִדְּשָׁנוּ בְּמִצְוֹתָיו וְצִוָּנוּ עַל מִצְוַת תְּפִלִּין.

פסוקי דזמרה

הוֹדוּ לַה' קִרְאוּ בִשְׁמוֹ, הוֹדִיעוּ בָעַמִּים עֲלִילוֹתָיו. שִׁירוּ לוֹ זַמְּרוּ לוֹ, שִׂיחוּ בְּכָל נִפְלְאוֹתָיו.

הסבר:

פסוקי דזמרה הם סדרת מזמורי תהילים הנאמרים בתחילת תפילת שחרית, המביעים שבח והודיה לה'.

קריאת שמע וברכותיה

שְׁמַע יִשְׂרָאֵל ה' אֱלֹהֵינוּ ה' אֶחָד. בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד.

תפילת העמידה

אֲדֹנָי שְׂפָתַי תִּפְתָּח וּפִי יַגִּיד תְּהִלָּתֶךָ. בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ, אֱלֹהֵי אַבְרָהָם, אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב, הָאֵל הַגָּדוֹל הַגִּבּוֹר וְהַנּוֹרָא, אֵל עֶלְיוֹן, גּוֹמֵל חֲסָדִים טוֹבִים, וְקוֹנֵה הַכֹּל, וְזוֹכֵר חַסְדֵי אָבוֹת, וּמֵבִיא גוֹאֵל לִבְנֵי בְנֵיהֶם לְמַעַן שְׁמוֹ בְּאַהֲבָה.

תחנון

וַיֹּאמֶר דָּוִד אֶל גָּד כִּי אֲנִי בְּצַר לִי מְאֹד נִפְּלָה נָּא בְיַד ה' כִּי רַבִּים רַחֲמָיו וּבְיַד אָדָם אַל אֶפֹּלָה.

הסבר:

תחנון הוא תפילת תחנונים הנאמרת לאחר תפילת העמידה, המבטאת את בקשת הרחמים מה'.

אשרי

אַשְׁרֵי יוֹשְׁבֵי בֵיתֶךָ, עוֹד יְהַלְלוּךָ סֶּלָה. אַשְׁרֵי הָעָם שֶׁכָּכָה לּוֹ, אַשְׁרֵי הָעָם שֶׁה' אֱלֹהָיו.

הסבר:

אשרי הוא מזמור תהילים הנאמר לאחר תפילת העמידה, המביע את האושר של יושבי בית ה' ואת השבח לה'.

ובא לציון

וּבָא לְצִיּוֹן גּוֹאֵל וּלְשָׁבֵי פֶשַׁע בְּיַעֲקֹב נְאֻם ה'. וַאֲנִי זֹאת בְּרִיתִי אוֹתָם אָמַר ה' רוּחִי אֲשֶׁר עָלֶיךָ וּדְבָרַי אֲשֶׁר שַׂמְתִּי בְּפִיךָ לֹא יָמוּשׁוּ מִפִּיךָ וּמִפִּי זַרְעֲךָ וּמִפִּי זֶרַע זַרְעֲךָ אָמַר ה' מֵעַתָּה וְעַד עוֹלָם.

הסבר:

ובא לציון הוא קטע הנאמר בסוף תפילת שחרית, המבטא את התקווה לגאולה ולשיבת ציון.

עלינו לשבח

עָלֵינוּ לְשַׁבֵּחַ לַאֲדוֹן הַכֹּל, לָתֵת גְּדֻלָּה לְיוֹצֵר בְּרֵאשִׁית, שֶׁלֹּא עָשָׂנוּ כְּגוֹיֵי הָאֲרָצוֹת, וְלֹא שָׂמָנוּ כְּמִשְׁפְּחוֹת הָאֲדָמָה, שֶׁלֹּא שָׂם חֶלְקֵנוּ כָּהֶם, וְגוֹרָלֵנוּ כְּכָל הֲמוֹנָם.

הסבר:

עלינו לשבח הוא תפילת סיום תפילת שחרית, המביעה את ההכרה בגדולת ה' ואת הייחוד של עם ישראל.
